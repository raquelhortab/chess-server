function main() {
  while( noTrayPresent() ){
    pickMove();
  }
  if(trayNotFull()){
    while(beepersInBag()){
      putBeeperInTray();
    }
  }
  while(noExitPresent()){
    pickMove();
  }
  exit();
}

function pickMove() {
    move();
    if( frontIsBlocked() ){
       turnRight();
    }
    if( frontIsBlocked() ){
       turnRight();
    }
    if( beepersPresent() ){
       pickBeeper();
    }
}
